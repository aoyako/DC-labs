package values;

import java.util.ArrayList;

import java.util.ArrayList;
import java.util.*;

public class Expression {
    Value value = new Value(0);
    Map<Integer, Unknown> variables = new HashMap<>();

    public Expression() {

    }

    public void add(Value v) {
        if (v.getType() == "value") {
            value.add(v);
        } else if (v.getType() == "variable") {
            Unknown var = (Unknown) v;
            if (var == null) {
                throw new RuntimeException("Cannot convert variable");
            }
            Unknown possible = variables.get(var.var_id);
            if (possible == null) {
                variables.put(var.var_id, var);
            } else {
                possible.multiply(var);
                variables.put(var.var_id, possible);
            }
        }
    }

    public void multiply(Value v) {
        value.multiply(v);
        for (Map.Entry<Integer, Unknown> var : variables.entrySet()) {
            var.getValue().multiply(v);
        }
    }

    public void divide(Value v) {
        value.divide(v);
        for (Map.Entry<Integer, Unknown> var : variables.entrySet()) {
            var.getValue().divide(v);
        }
    }

    public String toString() {
        String result = value.toString();

        for (Map.Entry<Integer, Unknown> var : variables.entrySet()) {
            result += " + " + var.getValue().toString();
        }
        return result;
    }
    
    public Expression(Expression e) {
        this.value = new Value(e.value);
        for (Map.Entry<Integer, Unknown> var : e.variables.entrySet()) {
            this.variables.put(var.getKey(), new Unknown(var.getValue()));
        }
    }

    public Value getValue() {
        return value;
    }
}
