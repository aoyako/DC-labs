package solver;

import values.*;

public class ThreeDiagonalSolver {
    protected SolverInterface solveMethod = null;

    public ThreeDiagonalSolver(SolverInterface solver) {
        solveMethod = solver;
    }

    public Expression[] solve(int[][] equation, int[] values) {
        Value[][] new_equation = new Value[equation.length][3];
        for (int i = 0; i < equation.length; i++) {
            for (int j = 0; j < 3; j++) {
                new_equation[i][j] = new Value(equation[i][j]);
            }
        }

        Value[] new_values = new Value[equation.length];
        for (int i = 0; i < equation.length; i++) {
            new_values[i] = new Value(values[i]);
        }

        return solveMethod.solve(new_equation, new_values);
    }
}
